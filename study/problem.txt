	function eventBind(elem, event, fn){
		// 大部分浏览器
		if(elem.addEventListener){
			elem.addEventListener(event.toLowerCase(), fn, false);
		}
		// IE
		else if(elem.attachEvent){
			elem.attachEvent('on'+event, fn);
		}
		// 其他情况
		else{
			elem['on' + event] = fn;
		}
	}
